## Learn（概念理解）

### サービス間の連携

複数のサービスが協調して動作する:

```python
class TaskService:
    def __init__(self, db: Session):
        self.db = db
        self.project_service = ProjectService(db)  # 他サービスを利用

    def create_task(self, task_data: TaskCreate) -> Task:
        # プロジェクトの存在確認（ProjectService を利用）
        project = self.project_service.get_project_by_id(task_data.project_id)
        if not project:
            raise ValueError(f"Project {task_data.project_id} not found")

        # タスク作成
        db_task = Task(**task_data.dict())
        self.db.add(db_task)
        self.db.commit()
        self.db.refresh(db_task)
        return db_task
```

**注意点**:
- サービスが互いに依存する場合、循環参照に注意
- DB セッションは共有する（同じトランザクション内）

### トランザクション管理

サービス層で**トランザクション境界**を管理する:

```python
def create_task_with_activity(self, task_data: TaskCreate) -> Task:
    """タスク作成 + 変更履歴の記録（1つのトランザクション）"""
    try:
        # タスク作成
        db_task = Task(**task_data.dict())
        self.db.add(db_task)
        self.db.flush()  # ID を取得するが commit はまだ

        # 変更履歴を記録
        activity = TaskActivity(
            task_id=db_task.id,
            field_name="status",
            new_value=task_data.status.value
        )
        self.db.add(activity)

        # 一括 commit
        self.db.commit()
        self.db.refresh(db_task)
        return db_task

    except Exception as e:
        self.db.rollback()
        raise
```

複雑な集計もサービス層に:

```python
class ProjectService:
    def get_project_summary(self, project_id: int) -> dict:
        """プロジェクトサマリー（集計ロジック）"""
        project = self.get_project_by_id(project_id)

        # タスク総数
        total_tasks = (
            self.db.query(func.count(Task.id))
            .filter(Task.project_id == project_id)
            .scalar()
        )

        # ステータス別集計
        status_results = (
            self.db.query(Task.status, func.count(Task.id))
            .filter(Task.project_id == project_id)
            .group_by(Task.status)
            .all()
        )
        tasks_by_status = {status.value: count for status, count in status_results}

        return {
            "project_id": project.id,
            "project_name": project.name,
            "total_tasks": total_tasks or 0,
            "tasks_by_status": tasks_by_status
        }
```

### ステータス遷移バリデーション

ビジネスルールもサービス層に:

```python
ALLOWED_STATUS_TRANSITIONS = {
    TaskStatus.TODO: [TaskStatus.IN_PROGRESS, TaskStatus.DONE],
    TaskStatus.IN_PROGRESS: [TaskStatus.DONE],
    TaskStatus.DONE: []
}

class TaskService:
    def update_task(self, task_id: int, task_data: TaskUpdate) -> Task:
        db_task = self.get_task_by_id(task_id)

        update_dict = task_data.dict(exclude_unset=True)

        # ステータス遷移バリデーション
        if "status" in update_dict:
            new_status = update_dict["status"]
            if not self._is_valid_transition(db_task.status, new_status):
                raise ValueError(
                    f"Invalid status transition: {db_task.status.value} → {new_status.value}"
                )

        # 変更履歴記録 + 更新
        for field, value in update_dict.items():
            old_value = getattr(db_task, field)
            if old_value != value:
                self._record_activity(task_id, field, old_value, value)
                setattr(db_task, field, value)

        self.db.commit()
        self.db.refresh(db_task)
        return db_task

    def _is_valid_transition(self, current: TaskStatus, new: TaskStatus) -> bool:
        if current == new:
            return True
        return new in ALLOWED_STATUS_TRANSITIONS.get(current, [])

    def _record_activity(self, task_id, field_name, old_value, new_value):
        activity = TaskActivity(
            task_id=task_id,
            field_name=field_name,
            old_value=str(old_value.value) if isinstance(old_value, Enum) else str(old_value),
            new_value=str(new_value.value) if isinstance(new_value, Enum) else str(new_value)
        )
        self.db.add(activity)
```

```python
    try:
        # 注文作成
        db_order = Order(user_id=order_data.user_id)
        self.db.add(db_order)
        self.db.flush()  # ID を取得

        # 明細作成
        for item_data in order_data.items:
            product = self.product_service.get_product_by_id(item_data.product_id)

            if product.stock < item_data.quantity:
                raise ValueError(f"Insufficient stock for {product.name}")

                order_id=db_order.id,
                product_id=item_data.product_id,
                quantity=item_data.quantity,
                price=product.price
            )
            self.db.add(db_item)

            # 在庫減少
            self.product_service.reduce_stock(product.id, item_data.quantity)

        # 合計金額計算
        db_order.total = sum(item.price * item.quantity for item in db_order.items)

        # 一括 commit
        self.db.commit()
        self.db.refresh(db_order)
        return db_order

        self.db.rollback()
```

```python
from services.project_service import ProjectService

@app.post("/projects", status_code=201)

    try:
        return service.create_task(task)
    except ValueError as e:
        raise HTTPException(status_code=422, detail=str(e))

@app.put("/tasks/{task_id}")
def update_task(task_id: int, task: TaskUpdate, db: Session = Depends(get_db)):
    service = TaskService(db)
    try:
        return service.update_task(task_id, task)
    except ValueError as e:
        if "not found" in str(e):
            raise HTTPException(status_code=404, detail=str(e))
```
