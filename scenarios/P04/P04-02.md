## Learn（概念理解）

### サービス層の実装パターン

```python
# services/user_service.py
from sqlalchemy.orm import Session
from models import User
from schemas import UserCreate, UserUpdate

class UserService:
    """ユーザー関連のビジネスロジック"""

    def __init__(self, db: Session):
        self.db = db

    def create_user(self, user_data: UserCreate) -> User:
        """ユーザー作成"""
        # 重複チェック
        if self._email_exists(user_data.email):
            raise ValueError("Email already exists")

        # DB操作
        db_user = User(
            name=user_data.name,
            email=user_data.email
        )
        self.db.add(db_user)
        self.db.commit()
        self.db.refresh(db_user)

        return db_user

    def get_user_by_id(self, user_id: int) -> User | None:
        """ID でユーザー取得"""
        return self.db.query(User).filter(User.id == user_id).first()

    def list_users(self) -> list[User]:
        """ユーザー一覧取得"""
        return self.db.query(User).order_by(User.created_at.desc()).all()

    def update_user(self, user_id: int, user_data: UserUpdate) -> User:
        """ユーザー更新"""
        db_user = self.get_user_by_id(user_id)
        if not db_user:
            raise ValueError("User not found")

        # 更新
        update_dict = user_data.dict(exclude_unset=True)
        for field, value in update_dict.items():
            setattr(db_user, field, value)

        self.db.commit()
        self.db.refresh(db_user)
        return db_user

    def delete_user(self, user_id: int):
        """ユーザー削除"""
        db_user = self.get_user_by_id(user_id)
        if not db_user:
            raise ValueError("User not found")

        self.db.delete(db_user)
        self.db.commit()

    def _email_exists(self, email: str) -> bool:
        """メールアドレスの重複チェック（内部メソッド）"""
        return self.db.query(User).filter(User.email == email).first() is not None
```

### ルートハンドラをリファクタリング

**Before（ファット）**:

```python
@app.post("/users")
def create_user(user: UserCreate, db: Session = Depends(get_db)):
    # 重複チェック
    existing = db.query(User).filter(User.email == user.email).first()
    if existing:
        raise HTTPException(status_code=409, detail="Email already exists")

    # DB操作
    db_user = User(name=user.name, email=user.email)
    db.add(db_user)
    try:
        db.commit()
    except IntegrityError:
        db.rollback()
        raise HTTPException(status_code=409, detail="Email conflict")

    db.refresh(db_user)
    return db_user
```

**After（薄い）**:

```python
@app.post("/users", status_code=201)
def create_user(user: UserCreate, db: Session = Depends(get_db)):
    service = UserService(db)
    try:
        return service.create_user(user)
    except ValueError as e:
        raise HTTPException(status_code=409, detail=str(e))
```

### エラーハンドリングの変換

**サービス層**: ビジネス例外（ValueError, KeyError など）を投げる
**ルートハンドラ**: ビジネス例外を HTTP エラーに変換

```python
# サービス層
def get_user_by_id(self, user_id: int) -> User:
    user = self.db.query(User).filter(User.id == user_id).first()
    if not user:
        raise ValueError(f"User with id {user_id} not found")  # ビジネス例外
    return user

# ルートハンドラ
@app.get("/users/{user_id}")
def get_user(user_id: int, db: Session = Depends(get_db)):
    service = UserService(db)
    try:
        return service.get_user_by_id(user_id)
    except ValueError as e:
        raise HTTPException(status_code=404, detail=str(e))  # HTTP エラーに変換
```

```python
# services/user_service.py
from sqlalchemy.orm import Session
from sqlalchemy.exc import IntegrityError
from models import User
from schemas import UserCreate, UserUpdate

class UserService:
    def __init__(self, db: Session):
        self.db = db

    def create_user(self, user_data: UserCreate) -> User:
        """ユーザー作成"""
        if self._email_exists(user_data.email):
            raise ValueError("Email already exists")

        db_user = User(name=user_data.name, email=user_data.email)
        self.db.add(db_user)

        try:
            self.db.commit()
        except IntegrityError:
            self.db.rollback()
            raise ValueError("Email conflict")

        self.db.refresh(db_user)
        return db_user

    def get_user_by_id(self, user_id: int) -> User:
        user = self.db.query(User).filter(User.id == user_id).first()
        if not user:
            raise ValueError(f"User with id {user_id} not found")
        return user

    def list_users(self) -> list[User]:
        return self.db.query(User).order_by(User.created_at.desc()).all()

    def update_user(self, user_id: int, user_data: UserUpdate) -> User:
        db_user = self.get_user_by_id(user_id)

        update_dict = user_data.dict(exclude_unset=True)

        # メール重複チェック（自分以外）
        if "email" in update_dict:
            existing = (
                self.db.query(User)
                .filter(User.email == update_dict["email"])
                .filter(User.id != user_id)
                .first()
            )
            if existing:
                raise ValueError("Email already in use")

        for field, value in update_dict.items():
            setattr(db_user, field, value)

        self.db.commit()
        self.db.refresh(db_user)
        return db_user

    def delete_user(self, user_id: int):
        db_user = self.get_user_by_id(user_id)
        self.db.delete(db_user)
        self.db.commit()

    def _email_exists(self, email: str) -> bool:
        return self.db.query(User).filter(User.email == email).first() is not None
```
