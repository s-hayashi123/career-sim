## Learn（概念理解）

### キャッシュとは

**計算結果を一時保存して再利用**:

```python
# ❌ Bad: 毎回計算
def get_summary(project_id):
    # 重い集計クエリ（50ms）
    return db.query(...).group_by(...).all()

# 1秒間に10リクエスト → 10回計算（500ms）

# ✅ Good: キャッシュ
cache = {}

def get_summary(project_id):
    if project_id in cache:
        return cache[project_id]  # 0.01ms

    result = db.query(...).group_by(...).all()  # 50ms
    cache[project_id] = result
    return result

# 1秒間に10リクエスト → 1回計算+9回キャッシュ（50ms + 0.09ms）
```

### インメモリキャッシュ（functools.lru_cache）

```python
from functools import lru_cache

@lru_cache(maxsize=128)
def get_project_summary(project_id: int) -> dict:
    # この関数の結果がキャッシュされる
    result = db.query(...).all()
    return result

# 同じproject_idで2回目以降は即座に返る
get_project_summary(1)  # 50ms（DB問い合わせ）
get_project_summary(1)  # 0.01ms（キャッシュ）
```

**注意**: DB セッションをキャッシュしない:

```python
# ❌ Bad: DB セッションをキャッシュ（セッションが古くなる）
@lru_cache
def get_summary(project_id, db):  # dbはキャッシュキーに含まれる
    ...

# ✅ Good: DB セッションは引数にしない
@lru_cache
def _compute_summary(project_id) -> dict:
    # 純粋な計算ロジックのみ
    ...

def get_summary(project_id, db):
    # キャッシュされた計算を使用
    return _compute_summary(project_id)
```

### キャッシュの無効化

データが更新されたら**キャッシュをクリア**:

```python
from functools import lru_cache

@lru_cache(maxsize=128)
def get_project_summary(project_id: int) -> dict:
    ...

def update_task(task_id, task_data, db):
    # タスク更新
    db_task = db.query(Task).filter(Task.id == task_id).first()
    db_task.status = task_data.status
    db.commit()

    # キャッシュを無効化
    get_project_summary.cache_clear()  # 全キャッシュクリア

    # または特定のキーだけクリア（cachetools使用）
    # cache.pop(db_task.project_id, None)
```

### TTL（Time To Live）付きキャッシュ

```python
from cachetools import TTLCache
import time

# 60秒で自動期限切れ
cache = TTLCache(maxsize=100, ttl=60)

def get_summary(project_id):
    if project_id in cache:
        return cache[project_id]

    result = compute_summary(project_id)
    cache[project_id] = result
    return result
```

### Redis（外部キャッシュ）

本格的なキャッシュ:

```python
import redis

r = redis.Redis(host='localhost', port=6379)

def get_summary(project_id):
    # キャッシュチェック
    cached = r.get(f"summary:{project_id}")
    if cached:
        return json.loads(cached)

    # 計算
    result = compute_summary(project_id)

    # キャッシュ保存（60秒TTL）
    r.setex(f"summary:{project_id}", 60, json.dumps(result))

    return result
```

### 課題: キャッシュを実装せよ

```python
def get_user_stats(user_id: int, db: Session) -> dict:
    """ユーザーの統計（重い計算）"""
    orders = db.query(Order).filter(Order.user_id == user_id).all()

    total_orders = len(orders)
    total_spent = sum(order.total for order in orders)
    avg_order = total_spent / total_orders if total_orders > 0 else 0

    return {
        "total_orders": total_orders,
        "total_spent": total_spent,
        "avg_order": avg_order
    }
```

**Q1.** lru_cache でキャッシュを追加せよ。
**Q2.** 注文更新時にキャッシュを無効化する関数を書け。

**A1. キャッシュ追加**:

```python
from functools import lru_cache
from sqlalchemy import func

@lru_cache(maxsize=128)
def _compute_user_stats(user_id: int) -> dict:
    """ユーザー統計の計算（キャッシュ対象）"""
    # DB セッションは外で取得
    db = SessionLocal()

    result = db.query(
        func.count(Order.id).label('total_orders'),
        func.sum(Order.total).label('total_spent'),
        func.avg(Order.total).label('avg_order')
    ).filter(Order.user_id == user_id).first()

    db.close()

    return {
        "total_orders": result.total_orders or 0,
        "total_spent": float(result.total_spent or 0),
        "avg_order": float(result.avg_order or 0)
    }

def get_user_stats(user_id: int, db: Session) -> dict:
    """ユーザー統計取得（キャッシュ使用）"""
    return _compute_user_stats(user_id)
```

```python
def update_order(order_id: int, order_data, db: Session):

    order.total = order_data.total
    db.commit()

    # キャッシュをクリア
    _compute_user_stats.cache_clear()
```

## Mission（本番課題）

### 状況

パフォーマンス改善が進み、ほとんどのエンドポイントが高速になりました。
自分で気づいて TL に提案:

> 「プロジェクトサマリーAPI（GET /projects/{id}/summary）は読み取り専用で、タスクが更新されない限り結果が変わりません。キャッシュを入れたらさらに高速化できると思います。」

TL からの返答:

> 「いいね！インメモリキャッシュで試してみよう。ただしタスク更新時にキャッシュを無効化することを忘れずに。」

### やること

1. **キャッシュ戦略を設計する**
   - 対象: `get_project_summary`
   - 方式: functools.lru_cache または cachetools.TTLCache
   - 無効化タイミング: タスク作成/更新/削除時

2. **サマリー計算をキャッシュ可能な関数に分離する**
   - DB セッションを引数にしない
   - 純粋な計算ロジック

3. **lru_cache を適用する**
   - maxsize=128
   - project_id をキーにキャッシュ

4. **キャッシュ無効化を実装する**
   - TaskService の create/update/delete でキャッシュクリア
   - `cache_clear()` を呼ぶ

5. **パフォーマンスを測定する**
   - Before: 50ms（SQL集計）
   - After（1回目）: 50ms
   - After（2回目以降）: 0.01ms

6. **performance-baseline.md を更新する**

7. **全テストが通ることを確認する**
   ```bash
   pytest tests/ -v
   ```

8. **PR を作成する**

```python
# services/project_service.py
from functools import lru_cache
from sqlalchemy import func

@lru_cache(maxsize=128)
def _compute_project_summary(project_id: int) -> dict:
    """プロジェクトサマリーを計算（キャッシュ対象）

    Note: DB セッションは引数にしない（キャッシュキーから除外）
    """
    db = SessionLocal()

    # プロジェクト取得
    project = db.query(Project).filter(Project.id == project_id).first()
    if not project:
        db.close()
        raise ValueError(f"Project {project_id} not found")

    # タスク総数
    total_tasks = (
        db.query(func.count(Task.id))
        .filter(Task.project_id == project_id)
        .scalar()
    ) or 0

    # ステータス別集計
    status_results = (
        db.query(Task.status, func.count(Task.id))
        .filter(Task.project_id == project_id)
        .group_by(Task.status)
        .all()
    )
    tasks_by_status = {status.value: count for status, count in status_results}

    # 優先度別集計
    priority_results = (
        db.query(Task.priority, func.count(Task.id))
        .filter(Task.project_id == project_id)
        .group_by(Task.priority)
        .all()
    )
    tasks_by_priority = {priority.value: count for priority, count in priority_results}

    db.close()

    return {
        "project_id": project.id,
        "project_name": project.name,
        "total_tasks": total_tasks,
        "tasks_by_status": tasks_by_status,
        "tasks_by_priority": tasks_by_priority
    }

class ProjectService:
    def get_project_summary(self, project_id: int) -> dict:
        """プロジェクトサマリー取得（キャッシュ使用）"""
        return _compute_project_summary(project_id)

# services/task_service.py
from services.project_service import _compute_project_summary

class TaskService:
    def create_task(self, task_data: TaskCreate) -> Task:
        # タスク作成
        db_task = Task(**task_data.dict())
        self.db.add(db_task)
        self.db.commit()
        self.db.refresh(db_task)

        # キャッシュ無効化
        _compute_project_summary.cache_clear()

        return db_task

    def update_task(self, task_id: int, task_data: TaskUpdate) -> Task:
        # タスク更新
        db_task = self.get_task_by_id(task_id)
        # ... 更新処理 ...
        self.db.commit()

        # キャッシュ無効化
        _compute_project_summary.cache_clear()

        return db_task

    def delete_task(self, task_id: int):
        db_task = self.get_task_by_id(task_id)
        self.db.delete(db_task)
        self.db.commit()

        # キャッシュ無効化
        _compute_project_summary.cache_clear()
```

- [ ] タスク作成/更新/削除時にキャッシュを無効化している
- [ ] パフォーマンスが改善されている（2回目以降）
- [ ] performance-baseline.md が更新されている

### 加点項目

- [ ] キャッシュヒット率を測定している
- [ ] TTL付きキャッシュを検討している
- [ ] Redis への移行パスを文書化している
- [ ] キャッシュの副作用（古いデータ）を認識している

### よくある間違い

- DB セッションをキャッシュしてしまう（セッションが古くなる）
- キャッシュ無効化を忘れる（古いデータを返す）
- 全キャッシュをクリアする（効率が悪い）
- キャッシュの効果を測定しない（改善確認ができない）
